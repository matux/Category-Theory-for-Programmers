func prodToSum<A, B, C>(_ t : (A, Either<B, C>)) -> Either<(A, B), (A, C)> {
  let (x, e) = t

    case let .right(z): return .right((x, z))
  }
}

    case let .right((x, z)):
        return (x, .right(z))

}
